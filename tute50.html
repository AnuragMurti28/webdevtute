This is JavaScript Tutorial | Strings and String Methods

INPUTS


var str = "This is a string";
console.log(str);
var position = str.indexOf('is'); //first occurance of a substring//
console.log(position);
var position2 = str.lastIndexOf('is'); //last occurence of a substring//
console.log(position2);
var substr = str.slice(1, 6); //substring - using slice(index,index)
console.log(substr);
var substr2 = str.substring(1, 6); //substring - using substring(index,index)
console.log(substr2);
var substr3 = str.substr(1, 7); //substring - using substr(index,length)
console.log(substr3);
var replaced = str.replace('string', 'harry'); //replace the string.
console.log(replaced);
console.log(str.toUpperCase()); //transform to uppercase.
console.log(str);
