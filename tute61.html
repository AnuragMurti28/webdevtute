// Printing the Functions from Math Object
let a = 34.65434;
let b = 49;

console.log("The value of a and b is", a, b);
console.log("The value of a and b rounded is ", Math.round(a), Math.round(b));
console.log("3 raised to the power 2 is ", Math.pow(3, 2));
console.log("square root of " + b + " is ", Math.sqrt(b));
console.log("square root of '50' is ", Math.sqrt(50));
console.log("Math.ceil() gives the rounded up to nearest integer value " + a + " ", Math.ceil(a));
console.log("Math.floor() gives the rounded down to nearest integer value " + a + " ", Math.floor(a));

console.log("Math.abs() gives absolute value of number ", Math.abs(-5.66));
console.log("Math.abs() gives absolute value of number ", Math.abs(5.66));

// Trignometric Functions
console.log("The value of sin(PI/2) is ", Math.sin(Math.PI / 2));
console.log("The value of cos(PI/2) is ", Math.cos(Math.PI / 2));
console.log("The value of tan(PI/2) is ", Math.tan(Math.PI / 2));

// min and max function

console.log("Minimum value of numbers 6, 7, 8 is ", Math.min(6, 7, 8));
console.log("Minimum value of numbers 16, 7, 18 is ", Math.min(16, 7, 18));
console.log("Maximum value of numbers 16, 7, 18 is ", Math.max(16, 7, 18));
console.log("Maximum value of numbers 16, 7, 18 is ", Math.max(16, 7, 18));

// Random number generator
let r = Math.random();
console.log("random number between 0 and 1: ", r);

// random number between (a1,b1) = a1 + (b1-a1)*Math.random()
let a1 = 1;
let b1 = 100;
let r1_100 = a1 + (b1 - a1) * Math.random();
console.log("The random number between 1 and 100 is", r1_100);
